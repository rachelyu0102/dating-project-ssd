    -- Create a database named DatingDB and use this script to build it.
	DROP TABLE Available;
    DROP TABLE ClientInterest;
	DROP TABLE Client;
    DROP TABLE Interest;
    DROP TABLE Location;
    GO

    CREATE TABLE Location
    (
    locationID INTEGER PRIMARY KEY IDENTITY(1,1),
    city VARCHAR(20),
    province VARCHAR(20),
    country VARCHAR(20),
    )

    GO
    CREATE TABLE Client(
	ID NVARCHAR(128) PRIMARY KEY,
    email NVARCHAR(256),
    userName NVARCHAR(256),
    birthdate DATE,
    gender VARCHAR(20),
    locationID INT,
    availableDate DATE NULL,
    CHECK(  gender IN ('male', 'female')),
    FOREIGN KEY (locationID) REFERENCES Location(locationID),
	FOREIGN KEY (ID) REFERENCES AspNetUsers(ID)
    );
    GO
    


    CREATE TABLE Available( 
	ID NVARCHAR(128) PRIMARY KEY,
    availableDate DATE NULL,
    timeStart TIME NULL,
    timeEnd TIME NULL,
	country VARCHAR(50) NULL,
	province VARCHAR(50) NULL,
	city VARCHAR(50) NULL,
    userName NVARCHAR(256),
    FOREIGN KEY (ID) REFERENCES Client(ID)
    );
    GO

    GO


    CREATE TABLE Interest
    (
    interest VARCHAR (20),
    PRIMARY KEY(interest)
    )

    GO

    CREATE TABLE ClientInterest
    (
	ID NVARCHAR(128),
    userName NVARCHAR(256),
    interest VARCHAR(20) FOREIGN KEY REFERENCES Interest(interest),
	FOREIGN KEY (ID) REFERENCES AspNetUsers(ID), 
	PRIMARY KEY (ID, interest)
    )

    
    GO
